// backend/routes/iaRoutes.js
import { fileURLToPath } from 'url';
import { dirname, join } from 'path';
import { readFileSync, writeFileSync } from 'fs';

const __filename = fileURLToPath(import.meta.url);
const __dirname = dirname(__filename);
const DATA_DIR = join(__dirname, '../data');

// PATHS
const PROFILES_PATH = join(DATA_DIR, 'ia_profiles.json');
const VIDEOS_PATH = join(DATA_DIR, 'ia_videos.json');

// Helper
function readJson(path) {
  const raw = readFileSync(path, 'utf-8');
  return JSON.parse(raw);
}

function writeJson(path, data) {
  writeFileSync(path, JSON.stringify(data, null, 2));
}

// DADOS INICIAIS (exemplo)

// ROTAS

export default function (app) {
  // Profiles
  app.get('/ia-profiles', (req, res) => {
    const profiles = readJson(PROFILES_PATH);
    res.json(profiles);
  });

  app.get('/ia-profiles/:id', (req, res) => {
    const { id } = req.params;
    const profiles = readJson(PROFILES_PATH);
    const profile = profiles.find(p => p.id === id);
    if (!profile) return res.status(404).json({ error: 'IA não encontrada' });
    res.json(profile);
  });

  // Videos
  app.get('/ia-videos', (req, res) => {
    const { iaId } = req.query;
    const videos = readJson(VIDEOS_PATH);
    const filtered = videos.filter(v => v.iaId === iaId);
    res.json(filtered);
  });

  // Etapa 2 – Missões de IA (backend skeleton)

// frontend/pages/FeedPage.jsx
import { useEffect, useState } from 'react';
import { Link } from 'react-router-dom';
import IaCard from '../components/IaCard';

export default function FeedPage() {
  const [ias, setIas] = useState([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetch('/api/ia-profiles')
      .then(res => res.json())
      .then(setIas)
      .finally(() => setLoading(false));
  }, []);

  if (loading) return <p>Carregando IAs...</p>;

  return (
    <div className="feed-page">
      <h1>Feed de IAs</h1>
      <div className="ia-list">
        {ias.length === 0 ? (
          <p>Nenhuma IA cadastrada ainda.</p>
        ) : (
          ias.map(ia => <IaCard key={ia.id} ia={ia} />)
        )}
      </div>
      <p><Link to="/">Voltar</Link></p>
    </div>
  );
}
3.5.3 frontend/pages/IaProfilePage.jsx
// frontend/pages/IaProfilePage.jsx
import { useParams, Link } from 'react-router-dom';
import { useState, useEffect } from 'react';
import VideoCard from '../components/VideoCard';

export default function IaProfilePage() {
  const { iaId } = useParams();
  const [ia, setIa] = useState(null);
  const [videos, setVideos] = useState([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    Promise.all([
      fetch(`/api/ia-profiles/${iaId}`).then(res => res.json()),
      fetch(`/api/ia-videos?iaId=${iaId}`).then(res => res.json())
    ]).then(([profile, vids]) => {
      setIa(profile);
      setVideos(vids);
    }).finally(() => setLoading(false));
  }, [iaId]);

  if (loading) return <p>Carregando...</p>;
  if (!ia) return <p>IA não encontrada.</p>;

  return (
    <div className="ia-profile-page">
      <h1>{ia.name} ({ia.type})</h1>
      <p>{ia.description}</p>
      <p>Desde: {new Date(ia.createdAt).toLocaleDateString()}</p>

      <h2>Vídeos Tutoriais</h2>
      <div className="video-grid">
        {videos.length === 0 ? (
          <p>Nenhum vídeo ainda.</p>
        ) : (
          videos.map(video => <VideoCard key={video.id} video={video} />)
        )}
      </div>

      <p><Link to="/feed">Voltar para o Feed</Link></p>
      <p><Link to={`/missions/${iaId}`}>Aprender a usar {ia.name}</Link></p>
    </div>
  );
}

// frontend/pages/VideoPage.jsx
import { useParams, Link } from 'react-router-dom';
import { useState, useEffect } from 'react';

export default function VideoPage() {
  const { videoId } = useParams();
  const [video, setVideo] = useState(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    // simula busca por todos os vídeos e filtra por id
    fetch(`/api/ia-videos`)
      .then(res => res.json())
      .then(videos => {
        const v = videos.find(v => v.id === videoId);
        setVideo(v || null);
      })
      .finally(() => setLoading(false));
  }, [videoId]);

  if (loading) return <p>Carregando vídeo...</p>;
  if (!video) return <p>Vídeo não encontrado.</p>;

  return (
    <div className="video-page">
      <h1>{video.title}</h1>
      <p>{video.description}</p>
      <p><strong>IA:</strong> {video.iaId}</p>

      <div className="video-player">
        {/* Em produção, você coloca um player real */}
        <p><strong>URL do vídeo:</strong> {video.videoUrl}</p>
      </div>

      <p><Link to={`/ia/${video.iaId}`}>Voltar ao perfil</Link></p>
    </div>
  );
}
